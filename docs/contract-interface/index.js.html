'use strict';
const ContractInterface = require('./libs/contract-interface')
var instanceOfInterface = {};

/**
 * Contract Interface's loader, load a singleton instance of ContractInterface
 * @param {String} net Network entity
 */
function loader(net) {
  if (typeof instanceOfInterface[net] === 'undefined') {
    instanceOfInterface[net] = {};
  }
  if (typeof instanceOfInterface[net]._instance === 'undefined') {
    instanceOfInterface[net]._instance = new ContractInterface(net);
    console.log(`ContractInterface loaded ${instanceOfInterface[net]._instance.version} at ${(new Date()).toString()}`);
  }
  return instanceOfInterface[net]._instance;
}

module.exports = loader;

if (typeof window !== 'undefined') {
  window.ContractInterface = loader;
}
